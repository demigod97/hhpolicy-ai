### Validation Function (Implementation)

```typescript
// File: src/services/embeddingValidator.ts

export interface EmbeddingValidationResult {
  valid: boolean;
  error?: string;
  warnings?: string[];
}

export function validateEmbedding(
  embedding: unknown,
  expectedDimension: number = 1536
): EmbeddingValidationResult {
  const warnings: string[] = [];

  // Check 1: Not null or undefined
  if (embedding === null || embedding === undefined) {
    return {
      valid: false,
      error: 'Invalid embeddings: null or undefined'
    };
  }

  // Check 2: Is array
  if (!Array.isArray(embedding)) {
    return {
      valid: false,
      error: `Invalid embeddings: expected array, got ${typeof embedding}`
    };
  }

  // Check 3: Not empty
  if (embedding.length === 0) {
    return {
      valid: false,
      error: 'Invalid embeddings: empty array'
    };
  }

  // Check 4: Correct dimensions
  if (embedding.length !== expectedDimension) {
    return {
      valid: false,
      error: `Invalid embedding dimensions: expected ${expectedDimension}, got ${embedding.length}`
    };
  }

  // Check 5: All values are numbers
  const hasNonNumeric = embedding.some(val => typeof val !== 'number' || isNaN(val));
  if (hasNonNumeric) {
    return {
      valid: false,
      error: 'Invalid embedding values: must be numeric (no NaN or non-numbers)'
    };
  }

  // Check 6: Values in expected range (warning, not error)
  const outOfRange = embedding.filter(val => val < -2 || val > 2);
  if (outOfRange.length > 0) {
    warnings.push(
      `${outOfRange.length} values outside typical range [-1, 1]: min=${Math.min(...embedding)}, max=${Math.max(...embedding)}`
    );
  }

  // Check 7: Not all zeros (suspicious)
  const allZeros = embedding.every(val => val === 0);
  if (allZeros) {
    warnings.push('All embedding values are zero - this is suspicious');
  }

  return {
    valid: true,
    warnings: warnings.length > 0 ? warnings : undefined
  };
}

// Usage in callback handler
export async function processDocumentCallback(callbackData: unknown) {
  try {
    const { source_id, chunks } = callbackData as any;

    // Validate all embeddings
    for (const chunk of chunks) {
      const validation = validateEmbedding(chunk.embedding);

      if (!validation.valid) {
        console.error(`Embedding validation failed for ${source_id}:`, validation.error);

        // Update source status to failed
        await supabase
          .from('sources')
          .update({
            processing_status: 'failed',
            error_message: `Embedding validation failed: ${validation.error}`
          })
          .eq('id', source_id);

        return { success: false, error: validation.error };
      }

      if (validation.warnings && validation.warnings.length > 0) {
        console.warn(`Embedding warnings for ${source_id}:`, validation.warnings);
      }
    }

    // Continue with normal processing...
    return { success: true };

  } catch (error) {
    return { success: false, error: error.message };
  }
}
```

```typescript
// File: supabase/functions/process-document/index.ts

const SIGNED_URL_TTL_SECONDS = 7200; // 2 hours (safe for most documents)
const MAX_PROCESSING_TIME_SECONDS = 3600; // 1 hour estimate

Deno.serve(async (req) => {
  try {
    const { source_id, file_path } = await req.json();

    // Generate signed URL with sufficient TTL
    const { data: signedUrlData, error: urlError } = await supabaseAdmin.storage
      .from('sources')
      .createSignedUrl(file_path, SIGNED_URL_TTL_SECONDS);

    if (urlError || !signedUrlData) {
      throw new Error(`Failed to generate signed URL: ${urlError?.message}`);
    }

    console.log(`Generated signed URL for ${source_id}, expires in ${SIGNED_URL_TTL_SECONDS}s`);

    // Call N8N webhook with signed URL
    const webhookUrl = Deno.env.get('DOCUMENT_PROCESSING_WEBHOOK_URL');
    const webhookPayload = {
      source_id,
      file_url: signedUrlData.signedUrl,
      file_path,
      callback_url: `${supabaseUrl}/functions/v1/process-document-callback`,
      metadata: {
        url_ttl_seconds: SIGNED_URL_TTL_SECONDS,
        generated_at: new Date().toISOString()
      }
    };

    const response = await fetch(webhookUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(webhookPayload)
    });

    if (!response.ok) {
      throw new Error(`N8N webhook failed: ${response.statusText}`);
    }

    return new Response(
      JSON.stringify({ success: true }),
      { status: 200 }
    );

  } catch (error) {
    console.error('Error processing document:', error);
    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500 }
    );
  }
});
```

### Callback Authentication (Implementation)

```typescript
// File: supabase/functions/process-document-callback/index.ts

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const N8N_AUTH_TOKEN = Deno.env.get('NOTEBOOK_GENERATION_AUTH'); // e.g., "Bearer secret-token"
const MAX_UNAUTHORIZED_ATTEMPTS = 10;
const RATE_LIMIT_WINDOW_MS = 60000; // 1 minute

// Simple in-memory rate limiting (for production, use Redis)
const unauthorizedAttempts = new Map<string, number[]>();

function validateAuthToken(authHeader: string | null): boolean {
  if (!authHeader) {
    return false;
  }

  // Compare with expected token
  return authHeader === N8N_AUTH_TOKEN;
}

function checkRateLimit(ip: string): boolean {
  const now = Date.now();
  const attempts = unauthorizedAttempts.get(ip) || [];

  // Clean old attempts outside window
  const recentAttempts = attempts.filter(time => now - time < RATE_LIMIT_WINDOW_MS);

  if (recentAttempts.length >= MAX_UNAUTHORIZED_ATTEMPTS) {
    return false; // Rate limited
  }

  // Record this attempt
  recentAttempts.push(now);
  unauthorizedAttempts.set(ip, recentAttempts);

  return true; // Not rate limited
}

Deno.serve(async (req) => {
  const clientIp = req.headers.get('x-forwarded-for') || 'unknown';

  try {
    // Step 1: Validate authentication
    const authHeader = req.headers.get('authorization');

    if (!validateAuthToken(authHeader)) {
      console.warn(`Unauthorized callback attempt from IP: ${clientIp}`);

      // Check rate limit
      if (!checkRateLimit(clientIp)) {
        console.error(`Rate limit exceeded for IP: ${clientIp}`);
        return new Response(
          JSON.stringify({ error: 'Too many unauthorized attempts' }),
          { status: 429, headers: { 'Content-Type': 'application/json' } }
        );
      }

      return new Response(
        JSON.stringify({ error: 'Unauthorized: Invalid authentication token' }),
        { status: 401, headers: { 'Content-Type': 'application/json' } }
      );
    }

    // Step 2: Parse and validate request body
    const callbackData = await req.json();
    const { source_id, extracted_text, chunks } = callbackData;

    if (!source_id || !chunks) {
      return new Response(
        JSON.stringify({ error: 'Invalid request: missing required fields' }),
        { status: 400, headers: { 'Content-Type': 'application/json' } }
      );
    }

    // Step 3: Process callback (validated and authenticated)
    console.log(`Processing authenticated callback for source: ${source_id}`);

    // ... Continue with normal processing ...

    return new Response(
      JSON.stringify({ success: true }),
      { status: 200, headers: { 'Content-Type': 'application/json' } }
    );

  } catch (error) {
    console.error('Error processing callback:', error);
    return new Response(
      JSON.stringify({ error: 'Internal server error' }),
      { status: 500, headers: { 'Content-Type': 'application/json' } }
```

### Callback Handler Logic (Implementation)

```typescript
// File: src/services/documentCallbackService.ts

export interface CallbackData {
  source_id: string;
  extracted_text?: string;
  chunks?: Array<{ text: string; embedding: number[]; chunk_index: number }>;
  metadata?: {
    is_scanned?: boolean;
    pages?: number;
    file_size_bytes?: number;
    suggested_actions?: string[];
  };
  error?: {
    code: string;
    message: string;
  };
}

export async function processDocumentCallback(callbackData: CallbackData) {
  const { source_id, extracted_text, chunks, metadata, error } = callbackData;

  try {
    // Handle error callbacks
    if (error) {
      await supabase
        .from('sources')
        .update({
          processing_status: 'failed',
          error_message: `${error.code}: ${error.message}`
        })
        .eq('id', source_id);

      return { success: false, error: error.message };
    }

    // Handle scanned documents (no text extracted)
    if (!extracted_text || extracted_text.trim().length === 0) {
      const isScanned = metadata?.is_scanned || false;

      if (isScanned) {
        // Scanned document - not an error, but needs user notification
        console.log(`Document ${source_id} is scanned - no text extracted`);

        await supabase
          .from('sources')
          .update({
            processing_status: 'completed',
            metadata: {
              ...metadata,
              no_text_extracted: true,
              is_scanned: true
            },
            processing_notes:
              'No text could be extracted from this document. ' +
              'It may contain scanned images without a text layer. ' +
              'Consider uploading a version with searchable text or enabling OCR processing.'
          })
          .eq('id', source_id);

        return {
          success: true,
          warnings: ['no_text_extracted'],
          message: 'Document processed but no text found (scanned images detected)'
        };
      } else {
        // Not scanned but still no text - this is suspicious
        await supabase
          .from('sources')
          .update({
            processing_status: 'completed',
            metadata: {
              ...metadata,
              no_text_extracted: true
            },
            processing_notes: 'No text could be extracted from this document.'
          })
          .eq('id', source_id);

        return {
          success: true,
          warnings: ['no_text_extracted'],
          message: 'Document processed but no text found'
        };
      }
    }

    // Normal processing with text
    if (chunks && chunks.length > 0) {
      // Store embeddings in vector database
      const documentsToInsert = chunks.map(chunk => ({
        source_id,
        content: chunk.text,
        embedding: chunk.embedding,
        chunk_index: chunk.chunk_index,
        metadata: {
          ...metadata,
          character_count: chunk.text.length
        }
      }));

      const { error: insertError } = await supabase
        .from('documents')
        .insert(documentsToInsert);

      if (insertError) {
        throw insertError;
      }
    }

    // Update source to completed
    await supabase
      .from('sources')
      .update({
        processing_status: 'completed',
        content: extracted_text,
        metadata: {
          ...metadata,
          chunk_count: chunks?.length || 0,
          character_count: extracted_text.length
        }
      })
      .eq('id', source_id);

    return { success: true };

  } catch (error) {
    console.error(`Error processing callback for ${source_id}:`, error);

    await supabase
      .from('sources')
      .update({
        processing_status: 'failed',
        error_message: error.message
      })
      .eq('id', source_id);

    return { success: false, error: error.message };
  }
}
```

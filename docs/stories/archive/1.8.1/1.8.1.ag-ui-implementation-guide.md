## Executive Summary

This guide provides step-by-step instructions for implementing AG-UI Protocol with CopilotKit in the PolicyAi application while maintaining full backward compatibility with the existing n8n webhook system and Supabase infrastructure.

### Implementation Strategy

**Dual-Mode Operation:**
- **Phase 1 (Days 1-5):** Run AG-UI alongside existing n8n system
- **Phase 2 (Days 6-10):** Gradual migration with feature flags
- **Phase 3 (Post-launch):** Full AG-UI with n8n as backup

### Key Benefits

✅ **Zero Downtime:** Existing chat continues working  
✅ **Gradual Migration:** Test AG-UI without breaking production  
✅ **Feature Flags:** Easy rollback if needed  
✅ **Enhanced UX:** Real-time streaming, better error handling  
✅ **Future-Proof:** Standard protocol, framework-agnostic

### Step 2: Create AG-UI Message Transformer (Day 1-2)

#### File: `src/utils/messageTransformers.ts`

```typescript
import { Message } from '@copilotkit/runtime-client-gql';
import { EnhancedChatMessage, Citation, MessageSegment } from '@/types/message';

/**
 * Transform n8n message format to AG-UI format
 */
export function transformN8nToAGUI(
  n8nMessage: any,
  sourceMap: Map<string, { id: string; title: string; type: string }>
): Message {
  const messageObj = n8nMessage.message;
  
  // Handle human messages
  if (messageObj?.type === 'human') {
    return {
      id: n8nMessage.id.toString(),
      role: 'user',
      content: typeof messageObj.content === 'string' 
        ? messageObj.content 
        : JSON.stringify(messageObj.content),
      createdAt: new Date().toISOString()
    };
  }
  
  // Handle AI messages with citations
  if (messageObj?.type === 'ai') {
    let content = '';
    let metadata: any = {};
    
    // Try to parse structured content
    if (typeof messageObj.content === 'string') {
      try {
        const parsed = JSON.parse(messageObj.content);
        if (parsed.output && Array.isArray(parsed.output)) {
          // Extract text and citations
          content = parsed.output.map((item: any) => item.text).join('\n\n');
          
          // Transform citations
          const citations: Citation[] = [];
          parsed.output.forEach((item: any, idx: number) => {
            if (item.citations) {
              item.citations.forEach((cite: any) => {
                const sourceInfo = sourceMap.get(cite.chunk_source_id);
                citations.push({
                  citation_id: idx + 1,
                  source_id: cite.chunk_source_id,
                  source_title: sourceInfo?.title || 'Unknown Source',
                  source_type: sourceInfo?.type || 'pdf',
                  chunk_lines_from: cite.chunk_lines_from,
                  chunk_lines_to: cite.chunk_lines_to,
                  chunk_index: cite.chunk_index,
                  excerpt: `Lines ${cite.chunk_lines_from}-${cite.chunk_lines_to}`
                });
              });
            }
          });
          
          metadata = { citations };
        } else {
          content = messageObj.content;
        }
      } catch (e) {
        content = messageObj.content;
      }
    } else if (typeof messageObj.content === 'object') {
      // Already structured
      if (messageObj.content.segments) {
        content = messageObj.content.segments
          .map((seg: any) => seg.text)
          .join('');
      }
      metadata = {
        citations: messageObj.content.citations || []
      };
    }
    
    return {
      id: n8nMessage.id.toString(),
      role: 'assistant',
      content,
      metadata,
      createdAt: new Date().toISOString()
    };
  }
  
  // Fallback
  return {
    id: n8nMessage.id.toString(),
    role: 'user',
    content: 'Unknown message format',
    createdAt: new Date().toISOString()
  };
}

/**
 * Transform AG-UI message to n8n format for storage
 */
export function transformAGUIToN8n(aguiMessage: Message): any {
  const isHuman = aguiMessage.role === 'user';
  
  return {
    type: isHuman ? 'human' : 'ai',
    content: aguiMessage.metadata?.citations 
      ? {
          segments: [{
            text: aguiMessage.content,
            citation_id: aguiMessage.metadata.citations.length > 0 ? 1 : undefined
          }],
          citations: aguiMessage.metadata.citations || []
        }
      : aguiMessage.content,
    additional_kwargs: aguiMessage.metadata?.additional_kwargs || {},
    response_metadata: aguiMessage.metadata?.response_metadata || {}
  };
}
```

### Step 3: Create AG-UI Chat Hook (Day 2)

#### File: `src/hooks/useAGUIChatMessages.tsx`

```typescript
import { useCopilotChat } from '@copilotkit/react-core';
import { useAuth } from '@/contexts/AuthContext';
import { useUserRole } from '@/hooks/useUserRole';
import { useQuery, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';
import { transformN8nToAGUI, transformAGUIToN8n } from '@/utils/messageTransformers';
import { useState, useEffect } from 'react';

export const useAGUIChatMessages = (notebookId?: string) => {
  const { user } = useAuth();
  const { role } = useUserRole();
  const queryClient = useQueryClient();
  
  // Get role-based instructions
  const instructions = getRoleBasedInstructions(role);
  
  // Initialize CopilotKit chat
  const {
    visibleMessages,
    appendMessage,
    isLoading: isCopilotLoading,
    stopGeneration,
    reset: resetCopilot
  } = useCopilotChat({
    instructions,
    makeSystemMessage: () => instructions
  });
  
  // Load existing messages from database
  const { data: dbMessages = [] } = useQuery({
    queryKey: ['chat-messages', notebookId],
    queryFn: async () => {
      if (!notebookId) return [];
      
      const { data, error } = await supabase
        .from('n8n_chat_histories')
        .select('*')
        .eq('session_id', notebookId)
        .order('id', { ascending: true });
      
      if (error) throw error;
      
      // Get sources for citation mapping
      const { data: sourcesData } = await supabase
        .from('sources')
        .select('id, title, type')
        .eq('notebook_id', notebookId);
      
      const sourceMap = new Map(sourcesData?.map(s => [s.id, s]) || []);
      
      // Transform to AG-UI format
      return data.map(msg => transformN8nToAGUI(msg, sourceMap));
    },
    enabled: !!notebookId && !!user
  });
  
  // Setup Realtime subscription
  useEffect(() => {
    if (!notebookId || !user) return;
    
    const channel = supabase
      .channel(`chat-${notebookId}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'n8n_chat_histories',
          filter: `session_id=eq.${notebookId}`
        },
        async (payload) => {
          console.log('New message from Realtime:', payload);
          
          // Get sources for transformation
          const { data: sourcesData } = await supabase
            .from('sources')
            .select('id, title, type')
            .eq('notebook_id', notebookId);
          
          const sourceMap = new Map(sourcesData?.map(s => [s.id, s]) || []);
          
          // Transform and add to messages
          const newMsg = transformN8nToAGUI(payload.new, sourceMap);
          
          queryClient.setQueryData(
            ['chat-messages', notebookId],
            (old: any[] = []) => {
              if (old.some(m => m.id === newMsg.id)) return old;
              return [...old, newMsg];
            }
          );
        }
      )
      .subscribe();
    
    return () => {
      supabase.removeChannel(channel);
    };
  }, [notebookId, user, queryClient]);
  
  // Send message function
  const sendMessage = async (content: string) => {
    if (!user || !notebookId) return;
    
    try {
      // Send via AG-UI
      await appendMessage({
        role: 'user',
        content
      });
      
      // Also send via n8n webhook for backup
      await supabase.functions.invoke('send-chat-message', {
        body: {
          session_id: notebookId,
          message: content,
          user_id: user.id
        }
      });
    } catch (error) {
      console.error('Error sending message:', error);
      throw error;
    }
  };
  
  return {
    messages: dbMessages,
    sendMessage,
    isLoading: isCopilotLoading,
    stopGeneration,
    reset: resetCopilot
  };
};

// Helper function for role-based instructions
function getRoleBasedInstructions(role: string): string {
  const instructions = {
    system_owner: "You are a strategic policy advisor for board-level executives. Focus on high-level insights, risk assessment, and strategic recommendations. Provide concise summaries with key takeaways.",
    
    company_operator: "You are a company operations assistant. Focus on practical implementation, operational compliance, and day-to-day policy application. Help users understand how policies affect their work.",
    
    executive: "You are an executive assistant. Provide concise, strategic insights and executive summaries. Focus on decision-support and key metrics that matter at the executive level.",
    
    administrator: "You are a comprehensive policy assistant with full system access. Help users navigate all policy documents, answer detailed questions, and provide in-depth analysis with citations.",
    
    policy_analyst: "You are a research-focused policy analyst assistant. Help with detailed policy analysis, document comparisons, and regulatory research. Provide thorough citations and evidence-based answers.",
    
    general_user: "You are a helpful policy assistant. Answer questions about accessible policy documents clearly and simply. Provide practical guidance on policy compliance."
  };
  
  return instructions[role as keyof typeof instructions] || instructions.general_user;
}
```

### Step 4: Create Dual-Mode Chat Component (Day 3)

#### File: `src/components/chat/AGUIChatInterface.tsx`

```typescript
import React, { useState } from 'react';
import { CopilotSidebar } from '@copilotkit/react-ui';
import { Button } from '@/components/ui/button';
import { Settings2 } from 'lucide-react';
import ChatArea from '@/components/notebook/ChatArea';
import { useAGUIChatMessages } from '@/hooks/useAGUIChatMessages';
import { useChatMessages } from '@/hooks/useChatMessages';

interface AGUIChatInterfaceProps {
  notebookId?: string;
  notebook?: any;
  onCitationClick?: (citation: any) => void;
}

export const AGUIChatInterface: React.FC<AGUIChatInterfaceProps> = ({
  notebookId,
  notebook,
  onCitationClick
}) => {
  // Feature flag: use AG-UI or fallback to n8n
  const [useAGUI, setUseAGUI] = useState(
    localStorage.getItem('use-ag-ui') === 'true'
  );
  
  // AG-UI hook
  const aguiChat = useAGUIChatMessages(notebookId);
  
  // Legacy n8n hook
  const legacyChat = useChatMessages(notebookId);
  
  // Choose which chat system to use
  const chat = useAGUI ? aguiChat : legacyChat;
  
  // Toggle between systems
  const toggleChatSystem = () => {
    const newValue = !useAGUI;
    setUseAGUI(newValue);
    localStorage.setItem('use-ag-ui', String(newValue));
    window.location.reload(); // Reload to reinitialize
  };
  
  return (
    <div className="relative h-full">
      {/* Feature toggle (dev only) */}
      {import.meta.env.DEV && (
        <div className="absolute top-4 right-4 z-50">
          <Button
            variant="outline"
            size="sm"
            onClick={toggleChatSystem}
            className="gap-2"
          >
            <Settings2 className="h-4 w-4" />
            {useAGUI ? 'Using AG-UI' : 'Using n8n'}
          </Button>
        </div>
      )}
      
      {/* Render appropriate chat interface */}
      {useAGUI ? (
        <CopilotSidebar
          defaultOpen={false}
          instructions={chat.instructions}
          labels={{
            title: "PolicyAi Assistant",
            initial: "How can I help you with policy compliance today?"
          }}
          className="h-full"
        >
          <ChatArea
            notebookId={notebookId}
            notebook={notebook}
            onCitationClick={onCitationClick}
            hasSource={true}
          />
        </CopilotSidebar>
      ) : (
        <ChatArea
          notebookId={notebookId}
          notebook={notebook}
          onCitationClick={onCitationClick}
          hasSource={true}
        />
      )}
    </div>
  );
};
```

```typescript
// Cache transformed messages
const messageCache = new Map<string, Message>();

const getCachedMessage = (id: string) => {
  if (!messageCache.has(id)) {
    const transformed = transformN8nToAGUI(rawMessage, sourceMap);
    messageCache.set(id, transformed);

  return messageCache.get(id);
};
```

# Story 1.19.5: User Settings Page (Basic)

**Epic**: 1.19 - Template Library & Enhanced Document Upload
**Status**: Approved
**Created**: 2025-11-28
**Priority**: MEDIUM
**Story Points**: 5
**Estimated Effort**: 2-3 days
**Dependencies**: None (independent story)

## Overview

Create a basic user settings page where users can view their profile information (email, role), edit their display name, change their password, and log out. This provides essential account management functionality while keeping the scope minimal for initial release.

---

## Business Requirements

**As a** PolicyAi user (any role)
**I want to** view and manage my account settings
**So that** I can update my display name, change my password, and understand my role permissions

**Value**:
- User autonomy for basic account management
- Reduced support requests for password resets
- Clear visibility into assigned role and permissions
- Professional user experience with account settings
- Foundation for future advanced settings (notifications, preferences, etc.)

---

## Current State

**No Settings Page Exists**: Users currently have no way to:
- View their assigned role
- Change their display name
- Change their password within the app
- See their profile information
- Access account management features

**Workarounds**:
- Password changes via Supabase Auth magic links (external)
- Role information not visible to users
- No display name personalization

### 2. Database Schema

**Leverage Existing Tables**:

```sql
-- Use existing auth.users table for email (read-only)
-- Use existing public.user_roles table for role (read-only)

-- Add display_name to public.profiles table (if not exists)
-- Or add new column to auth.users.raw_user_meta_data

-- Option 1: Create profiles table (recommended)
CREATE TABLE IF NOT EXISTS public.profiles (
    id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    display_name text,
    avatar_url text,
    created_at timestamptz DEFAULT now() NOT NULL,
    updated_at timestamptz DEFAULT now() NOT NULL
);

-- Enable RLS
ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

-- Policy: Users can view and update their own profile
CREATE POLICY "Users can view own profile"
    ON public.profiles
    FOR SELECT
    TO authenticated
    USING (auth.uid() = id);

CREATE POLICY "Users can update own profile"
    ON public.profiles
    FOR UPDATE
    TO authenticated
    USING (auth.uid() = id)
    WITH CHECK (auth.uid() = id);

-- Auto-create profile on user signup (trigger)
CREATE OR REPLACE FUNCTION public.create_profile_for_user()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.profiles (id, display_name)
    VALUES (NEW.id, NEW.email);
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER create_profile_on_signup
    AFTER INSERT ON auth.users
    FOR EACH ROW
    EXECUTE FUNCTION public.create_profile_for_user();
```

### 4. React Query Hooks

**File**: `src/hooks/useUserProfile.tsx`

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/integrations/supabase/client';
import { UserProfile, UpdateProfileRequest } from '@/types/userProfile';

// Fetch current user profile
export function useUserProfile() {
  return useQuery({
    queryKey: ['user-profile'],
    queryFn: async () => {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error('Not authenticated');

      // Get profile data
      const { data: profile, error: profileError } = await supabase
        .from('profiles')
        .select('*')
        .eq('id', user.id)
        .single();

      if (profileError) throw profileError;

      // Get user role
      const { data: roleData, error: roleError } = await supabase
        .from('user_roles')
        .select('role')
        .eq('user_id', user.id)
        .single();

      if (roleError) throw roleError;

      return {
        id: user.id,
        email: user.email!,
        display_name: profile?.display_name || user.email,
        avatar_url: profile?.avatar_url,
        role: roleData?.role || 'administrator',
        created_at: profile?.created_at,
        updated_at: profile?.updated_at,
      } as UserProfile;
    },
  });
}

// Update display name
export function useUpdateDisplayName() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (request: UpdateProfileRequest) => {
      const { data: { user } } = await supabase.auth.getUser();
      if (!user) throw new Error('Not authenticated');

      const { data, error } = await supabase
        .from('profiles')
        .update({ display_name: request.display_name })
        .eq('id', user.id)
        .select()
        .single();

      if (error) throw error;
      return data;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['user-profile'] });
    },
  });
}

// Change password via Supabase Auth
export function useChangePassword() {
  return useMutation({
    mutationFn: async (request: ChangePasswordRequest) => {
      const { error } = await supabase.auth.updateUser({
        password: request.newPassword,
      });

      if (error) throw error;
    },
  });
}
```
